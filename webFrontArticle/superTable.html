// 表格原理：整体分为head和body，复制table的thead放到head中（要复制高度和宽度），让head随table一起滚动，其实都是假象。
function tableplus(head,body,bodyH){
    $(body).find(".gridTable tbody tr").click(function(){
        $(this).css("background","#7cc5e5").siblings().css("background","");
    })
    var cloneTable = $(body).find(".gridTable").clone();//复制内容表格作为头部
    var gridTable=$(body).find(".gridTableHead").height();
    var headerDivWidth=$(body).width();//有时需去掉滚动条的宽度17就是固定头部div的宽度
    $(head).append(cloneTable[0]);//放到头部div中
    $(body).css("height",bodyH);
    $(head).css({
        "width":headerDivWidth,
        "height":gridTable,
        "margin-bottom":'-'+gridTable+"px",
    });
    $(body).scroll(function(){
        $(head).scrollLeft($(this).scrollLeft());
    })
}
